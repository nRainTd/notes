#include <iostream>
#include <initializer_list>

template <typename T>
class Link_list {
private:
  T data;
  Link_list<T>* next;

public:
  Link_list(T data) : data(data), next(nullptr) {}
  Link_list(T data, Link_list<T>* next) : data(data), next(next) {}
  Link_list(std::initializer_list<T> datas);
  ~Link_list();

public:
  T& operator[] (int index) {
    return this->get_node_by(index)->data;
  }
  const T& operator[] (int index) const {
    return this->get_node_by(index)->data;
  }

  void insert(T data, int index);
  void del(int index);

  void display();

private:
  Link_list* get_node_by(int index) {
    Link_list* tmp = this;

    for (int i = 0; i < index; ++i) {
      if (tmp->next == nullptr) {
        std::cout << "访问越界" << std::endl;
        exit(1);
      }
      tmp = tmp->next;
    };

    return tmp;
  }
};

// 构造函数和析构函数
template <typename T>
Link_list<T>::Link_list(std::initializer_list<T> datas) : next(nullptr) {
  if (datas.size() == 0) return;
  this->data = *datas.begin();

  auto tmp = this;
  for (auto p_data = datas.begin() + 1; p_data < datas.end(); ++p_data) {
    tmp->next = new Link_list<T>(*p_data);
    tmp = tmp->next;
  }
}
template <typename T>
Link_list<T>::~Link_list() {
  if (this->next != nullptr) delete this->next;
}


// 插入  
template <typename T>
void Link_list<T>::insert(T data, int index) {
  auto parent_node = this->get_node_by(index - 1);
  parent_node->next = new Link_list<T>(data, parent_node->next);
}

// 删除
template <typename T>
void Link_list<T>::del(int index) {
  auto parent_node = this->get_node_by(index - 1);
  auto to_be_del = parent_node->next;
  if (to_be_del == nullptr) return;

  parent_node->next = to_be_del->next;
  to_be_del->next = nullptr;

  delete to_be_del;
}

template <typename T>
void Link_list<T>::display() {
  auto tmp = this;

  while (tmp != nullptr) {
    std::cout << tmp->data << ' ';
    tmp = tmp->next;
  }

  std::cout << std::endl;
}

int main() {
  using namespace std;
  Link_list<int> list{ 1,2,3,4,5,6,7,8,9 };

  cout << "遍历输出:" << endl;
  list.display();

  cout << "把 6 改为 16" << endl;
  list[5] = 16;
  list.display();

  list.insert(11, 1);
  cout << "在 2 前面插入 11:" << endl;
  list.display();

  list.del(3);
  cout << "删除 3" << endl;
  list.display();

}
